200 inline Matrix3 AffineMicrosphere::compute_kirchhoff_stress(Matrix3 const& unimodular_F,
201  double const N) const
202 {
203  return unit_sphere.integrate(Matrix3::Zero().eval(),
204  [&](auto const& coordinates, auto const& l) -> Matrix3 {
205  auto const & [ r, r_outer_r ] = coordinates;
206 
207  // Deformed tangents
208  Vector3 const t = unimodular_F * r;
209 
210  // Microstretches
211  auto const micro_stretch = t.norm();
212 
213  return pade_first(micro_stretch, N) * t
214  * t.transpose();
215  });

219  double const N) const
220 {

222  [&](auto const& coordinates, auto const& l) -> CMatrix {

224 
225  // Deformed tangents

227
